### üì© data transformation pipeline
```r
## data transformation pipeline
mtcars %>%
  select(hp, wt, am)


mtcars %>%
  select(model, hp, wt, am) %>%
  filter(hp >= 200 & wt < 5 & am == "Auto")

  select(model, hp, wt, am) %>%
  filter(hp >= 200 | hp <= 90) %>%
  arrange(desc(hp))

mtcars %>%
  select(model, hp, wt, am) %>%
  filter(between (hp, 100, 180)) ## ‡πÅ‡∏£‡∏á‡∏°‡πâ‡∏≤ hp ‡∏≠‡∏¢‡∏π‡πà‡∏£‡∏∞‡∏´‡∏ß‡πà‡∏≤‡∏á 100 - 180
```
